## 一、项目概述

### 1.1 建设背景
作为国有大行科技子公司项目管理部，承担着总行各类IT系统实施的主体任务。为提升项目管理效能，基于OpenClaw+飞书生态，构建具备独立执行、自主闭环能力的PM数字员工，实现项目管理工作的智能化赋能。

### 1.2 建设目标
- **岗位化定位**：将数字员工作为虚拟PM岗位，能完整执行项目管理全流程业务
- **智能化协同**：基于飞书群聊实现项目级智能助手服务
- **权限隔离**：确保跨项目数据隔离，仅服务于成员所在项目
- **能力可编排**：通过Skill机制实现功能灵活上下架

#### 3.2.1 项目识别方案

```python
# 基于飞书群聊的项目绑定机制
class ProjectIdentifier:
    """项目识别器"""
    
    def identify_project(self, chat_id: str) -> Optional[str]:
        """
        通过群聊ID识别项目
        1. 查询群聊元数据中的项目绑定信息
        2. 若未绑定，引导管理员完成项目绑定
        """
        binding = self.db.query(
            "SELECT project_id FROM chat_project_binding WHERE chat_id = ",
            (chat_id,)
        )
        return binding['project_id'] if binding else None
    
    def verify_member_access(self, user_id: str, project_id: str) -> bool:
        """
        验证用户是否有权访问该项目
        1. 检查用户是否在项目群中
        2. 检查用户在项目系统中的角色
        """
        # 检查群成员资格
        is_chat_member = self.lark_client.check_chat_membership(
            chat_id=self.get_chat_by_project(project_id),
            user_id=user_id
        )
        
        # 检查项目成员资格
        is_project_member = self.db.query(
            "SELECT 1 FROM project_members WHERE user_id =  AND project_id = ",
            (user_id, project_id)
        )
        
        return is_chat_member and is_project_member
```

### 4.2 Skill标准化定义

```python
# skills/base_skill.py
from abc import ABC, abstractmethod
from pydantic import BaseModel

class SkillMetadata(BaseModel):
    """技能元数据"""
    skill_id: str
    name: str
    description: str
    category: str
    version: str
    required_permissions: List[str]
    input_schema: dict
    output_schema: dict

class BaseSkill(ABC):
    """技能基类"""
    
    def __init__(self, metadata: SkillMetadata):
        self.metadata = metadata
        self.is_enabled = True
        
    @abstractmethod
    async def execute(self, context: SkillContext) -> SkillResult:
        """执行技能逻辑"""
        pass
    
    async def pre_check(self, context: SkillContext) -> bool:
        """执行前检查"""
        # 权限检查
        if not self._check_permissions(context.user_id):
            raise PermissionError("用户无权限使用此技能")
        
        # 项目检查
        if not self._verify_project_access(context.project_id):
            raise AccessError("无权访问该项目")
            
        return True
    
    async def post_process(self, result: SkillResult) -> SkillResult:
        """结果后处理"""
        # 记录执行日志
        await self._log_execution(result)
        return result
```

# skills/daily_report_generator.py
class DailyReportGeneratorSkill(BaseSkill):
    """SK014: 日报/周报生成技能"""
    
    metadata = SkillMetadata(
        skill_id="SK014",
        name="日报周报生成",
        description="基于项目数据自动生成日报/周报",
        category="项目综合管理",
        version="1.0.0",
        required_permissions=["project.read", "report.create"],
        input_schema={
            "type": "object",
            "properties": {
                "report_type": {"type": "string", "enum": ["daily", "weekly"]},
                "date": {"type": "string", "format": "date"},
                "sections": {"type": "array", "items": {"type": "string"}}
            },
            "required": ["report_type"]
        },
        output_schema={
            "type": "object",
            "properties": {
                "report_content": {"type": "string"},
                "report_url": {"type": "string"}
            }
        }
    )
    
    async def execute(self, context: SkillContext) -> SkillResult:
        """执行日报生成"""
        
        # 1. 数据采集
        project_data = await self._collect_project_data(
            project_id=context.project_id,
            date=context.params.get('date', datetime.now())
        )
        
        # 2. 生成报告内容
        report_content = await self._generate_report_content(
            data=project_data,
            report_type=context.params['report_type']
        )
        
        # 3. 创建飞书文档
        doc_url = await self._create_lark_doc(
            title=f"{context.project_name} - {context.params['report_type']}报告",
            content=report_content
        )
        
        # 4. 发送通知
        await self._send_notification(
            chat_id=context.chat_id,
            doc_url=doc_url
        )
        
        return SkillResult(
            success=True,
            data={
                "report_content": report_content,
                "report_url": doc_url
            }
        )
    
    async def _collect_project_data(self, project_id: str, date: datetime) -> dict:
        """采集项目数据"""
        return {
            "completed_tasks": await self.db.query(
                "SELECT * FROM tasks WHERE project_id= AND status='completed' AND DATE(updated_at)=",
                (project_id, date)
            ),
            "ongoing_tasks": await self.db.query(
                "SELECT * FROM tasks WHERE project_id= AND status='in_progress'",
                (project_id,)
            ),
            "risks": await self.db.query(
                "SELECT * FROM risks WHERE project_id= AND status='open'",
                (project_id,)
            ),
            "cost_summary": await self.db.query(
                "SELECT SUM(amount) as total FROM costs WHERE project_id= AND DATE(cost_date)=",
                (project_id, date)
            )
        }
    
    async def _generate_report_content(self, data: dict, report_type: str) -> str:
        """使用LLM生成报告"""
        prompt = f"""
        请根据以下项目数据生成{report_type}报告：
        
        完成任务：{len(data['completed_tasks'])}项
        {json.dumps(data['completed_tasks'], ensure_ascii=False, indent=2)}
        
        进行中任务：{len(data['ongoing_tasks'])}项
        
        风险项：{len(data['risks'])}项
        
        成本支出：{data['cost_summary']['total']}元
        
        要求：
        1. 结构化输出，包含：工作概述、完成情况、风险提示、明日计划
        2. 突出重点和异常情况
        3. 使用Markdown格式
        """
        
        response = await self.llm.generate(prompt)
        return response.content

### 4.4 Skill动态上下架机制

```python
# skill_manager.py
class SkillManager:
    """技能管理器"""
    
    def __init__(self):
        self.skill_registry = {}
        self.enabled_skills = set()
        
    async def register_skill(self, skill: BaseSkill):
        """注册技能"""
        self.skill_registry[skill.metadata.skill_id] = skill
        await self._persist_skill_metadata(skill.metadata)
        
    async def enable_skill(self, skill_id: str, project_id: str = None):
        """上架技能"""
        if project_id:
            # 项目级启用
            await self.db.execute(
                "INSERT INTO project_skill_config (project_id, skill_id, enabled) VALUES (, , )",
                (project_id, skill_id, True)
            )
        else:
            # 全局启用
            self.enabled_skills.add(skill_id)
            
        await self._notify_skill_status_change(skill_id, "enabled")
        
    async def disable_skill(self, skill_id: str, project_id: str = None):
        """下架技能"""
        if project_id:
            await self.db.execute(
                "UPDATE project_skill_config SET enabled= WHERE project_id= AND skill_id=",
                (False, project_id, skill_id)
            )
        else:
            self.enabled_skills.discard(skill_id)
            
        await self._notify_skill_status_change(skill_id, "disabled")
        
    async def get_available_skills(self, project_id: str) -> List[BaseSkill]:
        """获取项目可用技能"""
        # 查询项目特定配置
        project_skills = await self.db.query(
            "SELECT skill_id FROM project_skill_config WHERE project_id= AND enabled=",
            (project_id, True)
        )
        
        # 合并全局启用技能
        available_skill_ids = set(project_skills) | self.enabled_skills
        
        return [self.skill_registry[sid] for sid in available_skill_ids]
```

# webhook_handler.py
from fastapi import FastAPI, Request, Header
from typing import Optional

app = FastAPI()

@app.post("/webhook/lark/message")
async def handle_lark_message(
    request: Request,
    x_lark_signature: Optional[str] = Header(None)
):
    """处理飞书消息事件"""
    
    # 1. 验证签名
    body = await request.body()
    if not verify_signature(body, x_lark_signature):
        return {"error": "Invalid signature"}
    
    # 2. 解析事件
    event = await request.json()
    
    # 3. 事件分发
    if event['header']['event_type'] == 'im.message.receive_v1':
        return await process_message_event(event)
    elif event['header']['event_type'] == 'approval.instance.create_v1':
        return await process_approval_event(event)
    
    return {"message": "ok"}

async def process_message_event(event: dict):
    """处理消息事件"""
    message = event['event']['message']
    sender = event['event']['sender']
    chat_id = message['chat_id']
    
    # 提取项目上下文
    project_id = await identify_project(chat_id)
    if not project_id:
        await send_lark_message(
            chat_id=chat_id,
            content="请先绑定项目：使用 /绑定项目 命令"
        )
        return
    
    # 权限验证
    if not await verify_member_access(sender['sender_id']['user_id'], project_id):
        await send_lark_message(
            chat_id=chat_id,
            content=" 您不是该项目成员，无权操作"
        )
        return
    
    # 构建上下文
    context = SkillContext(
        project_id=project_id,
        user_id=sender['sender_id']['user_id'],
        chat_id=chat_id,
        message_content=json.loads(message['content'])['text']
    )
    
    # 异步处理
    task_id = await submit_to_queue(context)
    
    # 立即响应

# task_queue.py
import aio_pika
import json

class TaskQueue:
    """基于RabbitMQ的任务队列"""
    
    def __init__(self):
        self.connection = None
        self.channel = None
        
    async def connect(self):
        self.connection = await aio_pika.connect_robust("amqp://localhost/")
        self.channel = await self.connection.channel()
        
    async def publish_task(self, task_type: str, payload: dict) -> str:
        """发布任务"""
        task_id = generate_task_id()
        
        message = {
            "task_id": task_id,
            "task_type": task_type,
            "payload": payload,
            "created_at": datetime.now().isoformat()
        }
        
        await self.channel.default_exchange.publish(
            aio_pika.Message(
                body=json.dumps(message).encode(),
                delivery_mode=aio_pika.DeliveryMode.PERSISTENT
            ),
            routing_key=f"task.{task_type}"
        )
        
        return task_id
    
    async def consume_tasks(self, task_type: str, callback):
        """消费任务"""
        queue = await self.channel.declare_queue(
            f"task.{task_type}",
            durable=True
        )
        
        async with queue.iterator() as queue_iter:
            async for message in queue_iter:
                async with message.process():
                    task_data = json.loads(message.body.decode())
                    await callback(task_data)

# 任务消费者
class TaskWorker:
    """任务处理worker"""
    
    async def process_skill_execution(self, task_data: dict):
        """处理技能执行任务"""
        context = SkillContext(**task_data['payload'])
        
        try:
            # 调用OpenClaw执行技能
            result = await openclaw_agent.execute_skill(context)
            
            # 发送结果到飞书
            await send_lark_message(
                chat_id=context.chat_id,
                content=format_skill_result(result)
            )
            
        except Exception as e:
            await send_lark_message(
                chat_id=context.chat_id,
                content=f" 执行失败: {str(e)}"

# skill_loader.py
import importlib
import os

class SkillLoader:
    """技能热加载器"""
    
    def __init__(self, skills_dir: str = "./skills"):
        self.skills_dir = skills_dir
        self.loaded_skills = {}
        self.file_watchers = {}
        
    async def load_skill(self, skill_module_name: str):
        """动态加载技能模块"""
        module_path = f"skills.{skill_module_name}"
        
        try:
            # 动态导入
            if module_path in self.loaded_skills:
                # 重新加载
                importlib.reload(self.loaded_skills[module_path])
            else:
                module = importlib.import_module(module_path)
                self.loaded_skills[module_path] = module
            
            # 实例化技能类
            skill_class = getattr(module, f"{skill_module_name.title()}Skill")
            skill_instance = skill_class()
            
            # 注册到技能管理器
            await skill_manager.register_skill(skill_instance)
            
            logger.info(f"Skill {skill_module_name} loaded successfully")
            
        except Exception as e:
            logger.error(f"Failed to load skill {skill_module_name}: {e}")
            raise
    
    async def watch_and_reload(self):
        """监听文件变化并热重载"""
        from watchdog.observers import Observer
        from watchdog.events import FileSystemEventHandler
        
        class SkillFileHandler(FileSystemEventHandler):
            def on_modified(self, event):
                if event.src_path.endswith('.py'):
                    skill_name = os.path.basename(event.src_path)[:-3]
                    asyncio.create_task(self.load_skill(skill_name))
        
        observer = Observer()
        observer.schedule(SkillFileHandler(), self.skills_dir, recursive=True)
        observer.start()

class AuditSystem:
    """审计系统"""
    
    async def audit_skill_executions(self, skill_id: str, sample_size: int = 100):
        """审计技能执行质量"""
        
        # 随机抽样
        samples = await db.query(
            """
            SELECT * FROM skill_execution_logs
            WHERE skill_id = 
            ORDER BY RAND()
            LIMIT 
            """,
            (skill_id, sample_size)
        )
        
        audit_results = []
        
        for sample in samples:
            # 验证输出正确性
            is_correct = await self._verify_output(
                skill_id=skill_id,
                input_params=sample['input_params'],
                output_result=sample['output_result']
            )
            
            audit_results.append({
                "log_id": sample['log_id'],
                "is_correct": is_correct
            })
        
        # 计算准确率
        accuracy_rate = sum(r['is_correct'] for r in audit_results) / len(audit_results)
        
        # 生成审计报告
        report = {
            "skill_id": skill_id,
            "audit_time": datetime.now().isoformat(),
            "sample_size": sample_size,
            "accuracy_rate": accuracy_rate,
            "details": audit_results
        }
        
        await save_audit_report(report)
        
        return report
