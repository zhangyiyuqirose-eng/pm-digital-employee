```python
class DocumentParseService:
    """
    文档解析核心服务.
    
    协调文件下载、解析、分类、提取、入库全流程。
    """
    
    def __init__(self, session: AsyncSession):
        self.session = session
        self.file_parser = FileParserService()
        self.classifier = DocumentClassifierService()
        self.extractor = DataExtractorService()
        self.importer = DataImportService(session)
        self.lark_client = get_lark_client()
    
    async def process_document(
        self,
        file_key: str,
        file_name: str,
        file_type: str,
        sender_id: str,
        chat_id: str,
        chat_type: str,
        message_id: str,
    ) -> DocumentParseResult:
        """
        处理文档完整流程.
        
        Args:
            file_key: 飞书文件Key
            file_name: 文件名
            file_type: 文件类型（file/image）
            sender_id: 发送者飞书ID
            chat_id: 会话ID
            chat_type: 会话类型（p2p/group）
            message_id: 飞书消息ID
            
        Returns:
            DocumentParseResult: 处理结果
        """
        # 创建解析记录
        parse_record = await self._create_parse_record(...)
        
        try:
            # 1. 文件下载
            file_info = await self._download_file(file_key)
            await self._update_status(parse_record, "file_downloaded")
            
            # 2. 文件解析
            content = await self.file_parser.parse(file_info)
            await self._update_status(parse_record, "content_parsed")
            
            # 3. 文档分类
            classification = await self.classifier.classify(
                content, file_info, user_context
            )
            await self._update_status(parse_record, "classified", classification)
            
            # 4. 信息提取
            extraction = await self.extractor.extract(
                content, classification.entity_types
            )
            await self._update_status(parse_record, "extracted", extraction)
            
            # 5. 置信度评估
            confidence = self._evaluate_confidence(extraction)
            
            # 6. 数据入库
            if confidence >= 0.8:
                result = await self.importer.import_all(
                    extraction, classification.project_id, user_context
                )
                await self._update_status(parse_record, "imported", result)
            else:
                await self._update_status(parse_record, "pending_confirmation")
            
            # 7. 结果反馈
            await self._send_result_card(parse_record)
            
            return DocumentParseResult(
                success=True,
                parse_record=parse_record,
                confidence=confidence,
            )
            
        except DocumentParseException as e:
            await self._handle_exception(parse_record, e)
            return DocumentParseResult(success=False, error=e)
```

class FileParserService:
    """
    文件解析服务.
    
    支持多种文档格式的文本提取。
    """
    
    SUPPORTED_EXTENSIONS = {
        # 文本文档
        "docx": self._parse_docx,
        "doc": self._parse_doc,
        "pdf": self._parse_pdf,
        "txt": self._parse_txt,
        "md": self._parse_txt,
        
        # 表格文档
        "xlsx": self._parse_xlsx,
        "xls": self._parse_xls,
        "csv": self._parse_csv,
        
        # 演示文档
        "pptx": self._parse_pptx,
        "ppt": self._parse_ppt,
        
        # 图片文档
        "jpg": self._parse_image,
        "jpeg": self._parse_image,
        "png": self._parse_image,
        "bmp": self._parse_image,
    }
    
    async def parse(self, file_info: FileInfo) -> ParsedContent:
        """
        解析文件内容.
        
        Args:
            file_info: 文件信息
            
        Returns:
            ParsedContent: 解析后的内容
        """
        ext = file_info.extension.lower()
        
        if ext not in self.SUPPORTED_EXTENSIONS:
            raise UnsupportedFormatError(f"不支持{ext}格式")
        
        parser = self.SUPPORTED_EXTENSIONS[ext]
        content = await parser(file_info.path)
        
        return ParsedContent(
            text=content.text,
            tables=content.tables,
            structure=content.structure,
            metadata=content.metadata,
        )
    
    async def _parse_docx(self, path: str) -> ParsedContent:
        """解析DOCX文件."""
        from docx import Document
        doc = Document(path)
        
        text = []
        tables = []
        
        for para in doc.paragraphs:
            text.append(para.text)
        
        for table in doc.tables:
            table_data = []
            for row in table.rows:
                row_data = [cell.text for cell in row.cells]
                table_data.append(row_data)
            tables.append(table_data)
        
        return ParsedContent(
            text="\n".join(text),
            tables=tables,
            structure={"type": "docx", "paragraphs": len(doc.paragraphs)},
        )
    
    async def _parse_pdf(self, path: str) -> ParsedContent:
        """解析PDF文件."""
        import pdfplumber
        
        text = []
        tables = []
        
        with pdfplumber.open(path) as pdf:
            for page in pdf.pages:
                text.append(page.extract_text() or "")
                page_tables = page.extract_tables()
                tables.extend(page_tables)
        
        return ParsedContent(
            text="\n".join(text),
            tables=tables,
            structure={"type": "pdf", "pages": len(pdf.pages)},
        )
    
    async def _parse_image(self, path: str) -> ParsedContent:
        """解析图片（OCR识别）."""
        # 调用OCR服务
        ocr_result = await self._call_ocr_service(path)
        
        return ParsedContent(
            text=ocr_result.text,
            tables=ocr_result.tables,
            structure={"type": "image", "ocr_confidence": ocr_result.confidence},
        )
```

class DocumentClassifierService:
    """
    文档分类服务.
    
    多级分类策略实现。
    """
    
    # 文件名关键词映射
    FILENAME_KEYWORDS = {
        "周报": ("weekly_report", "execution"),
        "会议纪要": ("meeting_minutes", "execution"),
        "WBS": ("wbs", "full_cycle"),
        "风险": ("risk_register", "full_cycle"),
        "里程碑": ("milestone_plan", "full_cycle"),
        "立项": ("initiation_doc", "initiation"),
        "验收": ("acceptance_report", "closing"),
        "总结": ("summary_report", "closing"),
    }
    
    async def classify(
        self,
        content: ParsedContent,
        file_info: FileInfo,
        user_context: UserContext,
    ) -> ClassificationResult:
        """
        执行文档分类.
        """
        # Step 1: 扩展名快速分类
        ext_category = self._classify_by_extension(file_info.extension)
        
        # Step 2: 文件名关键词匹配
        name_category = self._classify_by_filename(file_info.name)
        
        # Step 3: 内容深度分析（LLM）
        content_category = await self._classify_by_content(content.text)
        
        # Step 4: 项目关联推断
        project_match = await self._infer_project(content.text, user_context)
        
        # 综合判定
        return self._merge_classification(
            ext_category, name_category, content_category, project_match
        )
    
    async def _classify_by_content(self, text: str) -> ContentCategory:
        """使用LLM进行内容分类."""
        prompt = DOCUMENT_CLASSIFICATION_PROMPT.format(
            content_summary=text[:2000],
            key_content=self._extract_key_content(text),
        )
        
        response = await self.llm_gateway.generate(prompt, temperature=0.3)
        result = self._parse_classification_response(response.content)
        
        return ContentCategory(
            document_category=result["document_category"],
            project_phase=result["project_phase"],
            document_subtype=result["document_subtype"],
            confidence=result["confidence"],
            inferred_entity_types=result["inferred_entity_types"],
        )
    
    async def _infer_project(
        self,
        text: str,
        user_context: UserContext,
    ) -> ProjectMatch:
        """
        推断关联项目.
        
        策略：
        1. 群聊场景：优先使用群绑定项目
        2. 文档内容：匹配项目名称/编号
        3. 用户上下文：用户负责的项目
        """
        if user_context.chat_type == "group":
            # 群聊：使用群绑定项目
            binding = await self._get_group_binding(user_context.chat_id)
            if binding:
                return ProjectMatch(
                    project_id=binding.project_id,
                    match_type="group_binding",
                    confidence=1.0,
                )
        
        # 内容匹配
        project_keywords = self._extract_project_keywords(text)
        if project_keywords:
            matched_project = await self._match_project_by_keywords(project_keywords)
            if matched_project:
                return ProjectMatch(
                    project_id=matched_project.id,
                    match_type="content_match",
                    confidence=0.85,
                    keywords=project_keywords,
                )
        
        # 无法推断
        return ProjectMatch(
            project_id=None,
            match_type="unknown",
            confidence=0.0,
        )

    使用LLM从文档内容提取结构化数据。

    # 实体Schema定义

        },

            "priority": "enum: low/medium/high/critical",
            "start_date": "date: YYYY-MM-DD",
            "end_date": "date: YYYY-MM-DD",
            "assignee_name": "string",
            "progress": "int: 0-100",
        },
        "WeeklyReport": {
            "report_date": "date, required",
            "week_start": "date",
            "week_end": "date",
            "summary": "string",
            "completed_tasks": "array",
            "in_progress_tasks": "array",
            "next_week_plan": "string",
        },
        "MeetingMinutes": {
            "meeting_title": "string, required",
            "meeting_date": "date",
            "meeting_time": "string",
            "attendees": "array",
            "content": "string",
            "action_items": "array",
        },
        "Risk": {
            "title": "string, required",
            "category": "enum: schedule/cost/resource/...",
            "level": "enum: low/medium/high/critical",
            "probability": "int: 1-5",
            "impact": "int: 1-5",
            "owner_name": "string",
        },
    }
    
    async def extract(
        self,
        content: ParsedContent,
        entity_types: List[str],
        document_subtype: str,
    ) -> ExtractionResult:
        """
        执行数据提取.
        """
        # 选择专用Prompt
        if document_subtype == "weekly_report":
            prompt = WEEKLY_REPORT_EXTRACTION_PROMPT.format(
                document_content=content.text
            )
        elif document_subtype == "meeting_minutes":
            prompt = MEETING_MINUTES_EXTRACTION_PROMPT.format(
                document_content=content.text
            )
        else:
            prompt = DOCUMENT_EXTRACTION_PROMPT.format(
                entity_types=entity_types,
                entity_schema=self._build_schema_prompt(entity_types),
                document_content=content.text,
            )
        
        # LLM调用
        response = await self.llm_gateway.generate(
            prompt,
            temperature=0.2,
            max_tokens=4000,
        )
        
        # 解析结果
        extraction = self._parse_extraction_response(response.content)
        
        # 字段置信度计算
        field_confidences = self._calculate_field_confidences(extraction)
        
        return ExtractionResult(
            entities=extraction["extracted_entities"],
            overall_confidence=extraction["overall_confidence"],
            field_confidences=field_confidences,
            missing_fields=extraction["missing_required_fields"],
        )
    
    def _calculate_field_confidences(
        self,
        extraction: Dict,
    ) -> Dict[str, Dict[str, float]]:
        """
        计算各字段置信度.
        
        策略：
        - 明确标注的字段：95%
        - 格式正确的数值/日期：90%
        - 推断的字段：70%
        - 无法确定的字段：50%
        """
        confidences = {}
        
        for entity in extraction["extracted_entities"]:
            entity_conf = {}
            for field, value in entity["data"].items():
                if value is None:
                    entity_conf[field] = 0.0
                elif self._is_explicitly_marked(entity, field):
                    entity_conf[field] = 0.95
                elif self._is_valid_format(field, value):
                    entity_conf[field] = 0.90
                else:
                    entity_conf[field] = 0.70
            confidences[entity["entity_type"]] = entity_conf
        
        return confidences
```

```python
class DataImportService:
    """
    数据入库服务.
    
    调用现有Service层完成数据入库。
    """
    
    def __init__(self, session: AsyncSession):
        self.session = session
        self.validation_service = ValidationService()
        self.sync_engine = SyncEngine(session)
        self.access_control = AccessControlService(session)
    
    async def import_all(
        self,
        extraction: ExtractionResult,
        project_id: uuid.UUID,
        user_context: UserContext,
    ) -> ImportResult:
        """
        执行所有实体的入库.
        """
        imported_entities = []
        failed_entities = []
        conflicts = []
        
        for entity in extraction.entities:
            try:
                result = await self._import_single_entity(
                    entity, project_id, user_context
                )
                
                if result.status == "success":
                    imported_entities.append(result)
                elif result.status == "conflict":
                    conflicts.append(result)
                else:
                    failed_entities.append(result)
                    
            except Exception as e:
                failed_entities.append({
                    "entity_type": entity["entity_type"],
                    "error": str(e),
                })
        
        return ImportResult(
            imported_count=len(imported_entities),
            failed_count=len(failed_entities),
            conflict_count=len(conflicts),
            imported_entities=imported_entities,
            failed_entities=failed_entities,
            conflicts=conflicts,
        )
    
    async def _import_single_entity(
        self,
        entity: Dict,
        project_id: uuid.UUID,
        user_context: UserContext,
    ) -> SingleImportResult:
        """
        导入单个实体.
        """
        entity_type = entity["entity_type"]
        data = entity["data"]
        
        # 1. 数据校验
        validation_result = self.validation_service.validate_all(data, entity_type)
        if not validation_result.is_valid:
            return SingleImportResult(
                status="validation_failed",
                errors=validation_result.errors,
            )
        
        # 2. 权限检查
        has_permission = await self.access_control.check_data_write_permission(
            user_id=user_context.user_id,
            project_id=project_id,
            entity_type=entity_type,
        )
        if not has_permission:
            return SingleImportResult(status="permission_denied")
        
        # 3. 冲突检测
        existing = await self._find_existing_entity(entity_type, data, project_id)
        if existing:
            conflict = await self.sync_engine.detect_conflict(
                entity_type=entity_type,
                entity_id=existing.id,
                source_a=existing.data_source,
                data_a=self._entity_to_dict(existing),
                source_b="document_parse",
                data_b=data,
            )
            if conflict:
                return SingleImportResult(
                    status="conflict",
                    conflict_id=conflict.id,
                    existing_data=self._entity_to_dict(existing),
                    new_data=data,
                )
        
        # 4. 创建/更新实体
        if existing:
            # 更新
            entity_obj = await self._update_entity(existing, data)
        else:
            # 创建
            entity_obj = await self._create_entity(entity_type, data, project_id)
        
        # 5. 记录数据来源
        entity_obj.data_source = DataSource.DOCUMENT_PARSE
        entity_obj.sync_version = (existing.sync_version + 1) if existing else 1
        entity_obj.last_sync_at = datetime.now(timezone.utc)
        
        await self.session.commit()
        
        # 6. 记录版本
        await self.sync_engine.record_version(
            entity_type=entity_type,
            entity_id=entity_obj.id,
            operation="update" if existing else "create",
            data_after=self._entity_to_dict(entity_obj),
            data_source="document_parse",
        )
        
        return SingleImportResult(
            status="success",
            entity_id=entity_obj.id,
            entity_type=entity_type,
        )
```

```markdown

## 任务
分析文档内容，判断文档类型和关联的业务场景。

## 文档分类体系

SUPPORTED_FILE_MSG_TYPES = ["file", "image", "media"]

async def _handle_message_receive(
    event_data: Dict[str, Any],
    event_id: str,
    background_tasks: BackgroundTasks,
) -> Dict[str, Any]:
    """
    处理消息接收事件（扩展支持文件消息）.
    """
    message = event_data.get("message", {})
    msg_type = message.get("message_type", "")
    
    # 文本消息（现有）
    if msg_type == "text":
        return await _handle_text_message(...)
    
    # 文件消息（新增）
    if msg_type in SUPPORTED_FILE_MSG_TYPES:
        return await _handle_file_message(
            event_data, event_id, background_tasks
        )
    
    # 不支持的消息类型
    return {"code": 0, "msg": "unsupported_message_type"}


async def _handle_file_message(
    event_data: Dict[str, Any],
    event_id: str,
    background_tasks: BackgroundTasks,
) -> Dict[str, Any]:
    """
    处理文件消息.
    """
    message = event_data.get("message", {})
    sender = event_data.get("sender", {})
    
    content = json.loads(message.get("content", "{}"))
    file_key = content.get("file_key", "")
    file_name = content.get("name", "unknown")
    file_type = message.get("message_type", "file")
    file_size = content.get("size", 0)
    
    # 文件大小检查
    if file_size > 50 * 1024 * 1024:
        return {"code": -1, "msg": "文件大小超过限制（50MB）"}
    
    sender_id = sender.get("sender_id", {}).get("open_id", "")
    chat_id = message.get("chat_id", "")
    chat_type = message.get("chat_type", "p2p")
    message_id = message.get("message_id", "")
    
    # 异步处理
    background_tasks.add_task(
        _process_file_async,
        file_key=file_key,
        file_name=file_name,
        file_type=file_type,
        sender_id=sender_id,
        chat_id=chat_id,
        chat_type=chat_type,
        message_id=message_id,
        event_id=event_id,
    )
    
    # 立即返回
    return {"code": 0, "msg": "文件已接收，正在处理..."}


async def _process_file_async(
    file_key: str,
    file_name: str,
    file_type: str,
    sender_id: str,
    chat_id: str,
    chat_type: str,
    message_id: str,
    event_id: str,
) -> None:
    """
    异步处理文件解析.
    """
    # 幂等检查
    idempotency = FileMessageIdempotencyService()
    if await idempotency.is_processed(event_id):
        logger.warning(f"File message already processed: {event_id}")
        return
    
    # 发送处理中提示
    lark_service = get_lark_service()
    await lark_service.send_text_message(
        chat_id=chat_id,
        text=f"📄 正在处理文档《{file_name}》...",
    )
    
    # 执行解析
    try:
        parse_service = DocumentParseService(get_session())
        result = await parse_service.process_document(
            file_key=file_key,
            file_name=file_name,
            file_type=file_type,
            sender_id=sender_id,
            chat_id=chat_id,
            chat_type=chat_type,
            message_id=message_id,
        )
        
        # 标记已处理
        await idempotency.mark_processed(event_id, result)
        
    except Exception as e:
        logger.error(f"Document parse failed: {e}")
        await lark_service.send_text_message(
            chat_id=chat_id,
            text=f"❌ 文档处理失败：{str(e)}",
        )

## 八、权限控制方案

### 8.1 文档处理权限

```python
class DocumentParsePermissionChecker:
    """文档解析权限校验."""
    
    async def check_parse_permission(
        self,
        user_context: UserContext,
        file_info: FileInfo,
    ) -> PermissionCheckResult:
        """
        权限检查流程：
        1. 文件类型白名单检查
        2. 文件大小限制检查
        3. 项目成员身份检查
        4. 数据写入权限检查
        """
        # 白名单检查
        if file_info.extension not in SUPPORTED_EXTENSIONS:
            return PermissionCheckResult(
                allowed=False,
                reason=f"不支持{file_info.extension}格式文件",
            )
        
        # 大小检查
        if file_info.size > MAX_FILE_SIZE:
            return PermissionCheckResult(
                allowed=False,
                reason="文件大小超过50MB限制",
            )
        
        # 项目成员检查
        if user_context.current_project:
            access_control = AccessControlService(self._session)
            has_write = await access_control.check_permission(
                user_id=user_context.user_id,
                project_id=user_context.current_project,
                action=PermissionAction.WRITE,
            )
            if not has_write:
                return PermissionCheckResult(
                    allowed=False,
                    reason="您没有该项目的数据写入权限",
                )
        
        return PermissionCheckResult(allowed=True)
```

### 8.2 项目关联校验

```python
async def validate_project_association(
    inferred_project_id: uuid.UUID,
    user_context: UserContext,
    chat_type: str,
) -> ProjectAssociationResult:
    """
    项目关联校验规则：
    1. 群聊：项目必须与群绑定一致
    2. 单聊：用户必须为项目成员
    3. 无法推断：需用户选择
    """
    # 群聊场景
    if chat_type == "group":
        binding = await get_group_binding(user_context.chat_id)
        if binding and binding.project_id != inferred_project_id:
            return ProjectAssociationResult(
                valid=False,
                reason="推断项目与群绑定不一致",
                suggested_project=binding.project_id,
            )
    
    # 无法推断项目
    if not inferred_project_id:
        projects = await get_user_projects(user_context.user_id)
        return ProjectAssociationResult(
            valid=False,
            requires_selection=True,
            candidate_projects=projects,
        )
    
    # 用户项目成员检查
    is_member = await check_project_membership(
        user_context.user_id, inferred_project_id
    )
    if not is_member:
        return ProjectAssociationResult(
            valid=False,
            reason="您不是该项目的成员",
        )
    
    return ProjectAssociationResult(valid=True, project_id=inferred_project_id)
```

## 九、风险评估与应对

### 9.1 技术风险

| 风险项 | 等级 | 应对措施 |
|--------|------|----------|
| LLM输出不稳定 | 高 | 多次解析对比 + 结构化输出强制 + 人工兜底 |
| 文件解析失败 | 中 | 格式兼容性测试 + 失败提示 + 手动录入 |
| OCR准确率低 | 中 | 高质量OCR服务 + 置信度阈值提高 + 人工确认 |
| 文件下载超时 | 低 | 重试机制 + 异步处理 + 用户重发 |

### 9.2 数据风险

| 风险项 | 等级 | 应对措施 |
|--------|------|----------|
| 数据提取错误 | 高 | 置信度评估 + 用户确认 + 数据版本控制 |
| 数据冲突 | 中 | 冲突检测 + 冲突解决卡片 + 版本历史 |
| 项目关联错误 | 中 | 多因素推断 + 用户确认 + 群绑定校验 |
| 数据覆盖 | 高 | 版本控制 + 软删除 + 操作日志 |

### 9.3 安全风险

| 风险项 | 等级 | 应对措施 |
|--------|------|----------|
| 恶意文件上传 | 中 | 文件类型白名单 + 大小限制 + 内容安全检查 |
| 敏感信息泄露 | 高 | 数据脱敏 + 权限控制 + 安全审计 |
| Prompt注入 | 高 | SafetyGuard检测 + 内容过滤 + 输出安全检查 |
